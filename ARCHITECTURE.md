## Data Model

```typescript
CourseWeek {
  weekNumber: 1-10
  title: string
  description: string
  focusArea: string
  lessons: Lesson[]
}

Lesson {
  id: string
  dayNumber: 1-7
  title: string
  content: string
  scriptureReference?: string
  prayerFocus?: string
  nextSteps?: string[]
}

UserProgress {
  userId: string
  weekNumber: number
  completedLessons: string[] (lesson IDs)
  marks: Mark[]
}

JournalEntry {
  id: string
  userId: string
  date: Date
  weekNumber: number
  lessonId: string
  content: string
  prayerRequest?: string
  breakthrough?: string
}

UserNotes {
  id: string
  userId: string
  lessonId: string
  content: string
  createdAt: Date
  updatedAt: Date
}
```
